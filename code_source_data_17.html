Timing of clonal expansions in endometrial glands

Markdown file to present the methods to estimate timing of clonal expansions in endometrial glands. Before you start the following analyses, please convert the corresponding excel source data files to tab-delimited text files. The source data files are provided in our paper (Yamaguchi et al. Nat Commun in press).

Estimation of timing of clonal expansions in endometrial glands of subject 36

mutData$total <- mutData$num_public + mutData$num_partially_shared + mutData$num_private
mutData$se <- sqrt(mutData$prop_public * (1 - mutData$prop_public) / mutData$total)
mutData$lower_95 <- mutData$prop_public - 1.96 * mutData$se
mutData$upper_95 <- mutData$prop_public + 1.96 * mutData$se

sample_age = 50 + 5/12
cluster <- c("cluster 1", "cluster 3")
age <- vector(mode="numeric", length=length(cluster))
lower_95 <- vector(mode="numeric", length=length(cluster))
upper_95 <- vector(mode="numeric", length=length(cluster))

for (i in 1:length(cluster)) {
  temp_data <- mutData[mutData$cluster == cluster[i], ]
  age[i] <- sample_age * mean(temp_data$prop_public)
  lower_95[i] <- sample_age * mean(temp_data$lower_95)
  upper_95[i] <- sample_age * mean(temp_data$upper_95)
}

res <- data.frame(
  Cluster=cluster,
  Age=age,
  Lower_95=lower_95,
  Upper_95=upper_95)

# Timing of expansions of recent ancestral clones giving rise to two clusters of glands

Visualize timing of clonal expansions in endometrial glands of subject 36

Glands in cluster 1

mutData2 <- mutData[, c("cluster", "sample", "prop_public", "prop_partially_shared", "prop_private")]
mutData_long <- mutData2 %>% gather(type, proportion, -cluster, -sample, factor_key=TRUE)
mutData_long$sample2 <- sub("-WGS", "", mutData_long$sample)
mutData_long$sample3 <- factor(mutData_long$sample2, 
  levels=rev(c("S36-AIV-8", "S36-B-5", "S36-C-10", "S36-C-13", "S36-C-15", "S36-D-1", "S36-D-2", "S36-D-4", "S36-D-18", "S36-D-20")))
mutData_long$type2 <- sub("prop_", "", mutData_long$type)
mutData_long$type3 <- factor(mutData_long$type2, levels=c("private", "partially_shared", "public"))

mutData_long_cl1 <- mutData_long[mutData_long$cluster == "cluster 1", ]
mutData_long_cl3 <- mutData_long[mutData_long$cluster == "cluster 3", ]
ave_cl1 <- mean(mutData_long_cl1[mutData_long_cl1$type2 == "public", ]$proportion)
ave_cl3 <- mean(mutData_long_cl3[mutData_long_cl3$type2 == "public", ]$proportion)

my_colors <- c("#5a0b8e", "#b57303", "#a03910")

p_cl1 <- ggplot(data=mutData_long_cl1, aes(x=sample3, y=proportion, fill=type3))
p_cl1 <- p_cl1 + geom_bar(stat="identity", position="fill")
p_cl1 <- p_cl1 + geom_hline(yintercept=ave_cl1, linetype="dashed")
p_cl1 <- p_cl1 + coord_flip()
p_cl1 <- p_cl1 + scale_y_continuous(sec.axis=sec_axis(~ . * sample_age, name="Age"))
p_cl1 <- p_cl1 + scale_fill_manual(values=my_colors, name="", 
  breaks=c("public", "partially_shared", "private"),
  labels=c("Public", "Partially shared", "Private"))
p_cl1 <- p_cl1 + xlab("")
p_cl1 <- p_cl1 + ylab("Proportion of somatic SNVs")
p_cl1 <- p_cl1 + theme(
  axis.ticks=element_blank(),
  axis.text.x=element_text(size=20, colour="black"),
  axis.text.y=element_text(size=20, colour="black"),
  axis.title=element_text(size=20),
  legend.text=element_text(size=20),
  panel.grid.major=element_blank(),
  panel.border=element_blank(),
  panel.background=element_blank(),
  legend.position="top")

plot(p_cl1)
