Plots of characteristics of cancer-associated gene mutations in endometrial glands

Markdown file to present the methods to generate plots of characteristics of cancer-associated gene mutations in endometrial glands. Before you start the following analyses, please convert the corresponding excel source data files to tab-delimited text files. The source data files are provided in our paper (Yamaguchi et al. Nat Commun in press).

Visualize bar plots

mutProp_long <- mutProp %>% gather(type, proportion, -gene, factor_key=TRUE)
mutProp_long$gene2 <- factor(mutProp_long$gene, levels=rev(mutProp$gene))
mutProp_long$type2 <- factor(mutProp_long$type, levels=c("both", "indel", "SNV"))

p_bar <-ggplot(data=mutProp_long, aes(x=gene2, y=proportion, fill=type2)) 
p_bar <- p_bar + geom_hline(aes(yintercept=0.1), colour="black", linetype="dashed")
p_bar <- p_bar + geom_hline(aes(yintercept=0.2), colour="black", linetype="dashed")
p_bar <- p_bar + geom_bar(width=0.9, stat="identity", colour="black")
p_bar <- p_bar + coord_flip()
p_bar <- p_bar + scale_y_continuous(breaks=c(0, 0.1, 0.2), labels=c(0, 0.1, 0.2))
p_bar <- p_bar + scale_fill_manual(
  name="",
  values=rev(c("white", "darkgray", "black")),
  breaks=rev(c("both", "indel", "SNV")),
  labels=rev(c("Both", "Indel", "SNV")))
p_bar <- p_bar + theme(
  axis.ticks=element_blank(),
  axis.text.x=element_text(size=20, colour="black"),
  axis.text.y=element_text(size=20, colour="black", face="italic"),
  axis.title=element_text(size=20),
  panel.grid.major=element_blank(),
  panel.border=element_blank(),
  panel.background=element_blank(),
  legend.text=element_text(size=20),
  legend.direction="horizontal",
  legend.position="bottom")
p_bar <- p_bar + xlab("")
p_bar <- p_bar + ylab("Proportion of glands with mutation")

plot(p_bar)
